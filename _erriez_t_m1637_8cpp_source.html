191 uint8_t TM1637::getKeys()
192 {
193  uint8_t keyCode;
194 
195  start();
196  writeByte(TM1637_CMD_DATA | TM1637_DATA_READ_KEYS);
197  start();
198  keyCode = readByte();
199  stop();
200 
201  // Check if key is down (at least one bit is zero)
202  if (keyCode != 0xFF) {
203  // Invert keyCode:
204  // Bit | 7 6 5 4 3 2 1 0
205  // ------+-------------------------
206  // From | S0 S1 S2 K1 K2 1 1 1
207  // To | S0 S1 S2 K1 K2 0 0 0
208  keyCode = ~keyCode;
209 
210  // Shift bits to:
211  // Bit | 7 6 5 4 3 2 1 0
212  // ------+------------------------
213  // To | 0 0 0 0 K2 S2 S1 S0
214  keyCode = (uint8_t)((keyCode & 0x80) >> 7 |
215  (keyCode & 0x40) >> 5 |
216  (keyCode & 0x20) >> 3 |
217  (keyCode & 0x08));
218  }
219 
220  return keyCode;
221 }

276 void TM1637::writeByte(uint8_t data)
277 {
278  for (uint8_t i = 0; i < 8; i++) {
279  TM1637_PIN_DELAY();
280  TM1637_CLK_LOW();
281 
282  if (data & 0x01) {
283  TM1637_DIO_HIGH();
284  } else {
285  TM1637_DIO_LOW();
286  }
287  data >>= 1;
288 
289  TM1637_PIN_DELAY();
290  TM1637_CLK_HIGH();
291  }
292 
293  TM1637_PIN_DELAY();
294  TM1637_CLK_LOW();
295 
296  // Set DIO pin to input with pull-up to read ACK
297  TM1637_DIO_HIGH();
298  TM1637_DIO_INPUT();
299  TM1637_PIN_DELAY();
300 
301  TM1637_CLK_HIGH();
302  TM1637_PIN_DELAY();
303  // Skip reading DIO pin: ack = _dioPin;
304  TM1637_CLK_LOW();
305  TM1637_PIN_DELAY();
306 
307  // Set DIO pin to output
308  TM1637_DIO_OUTPUT();
309  TM1637_PIN_DELAY();
310 }

316 uint8_t TM1637::readByte()
317 {
318  uint8_t retval = 0;
319 
320  // Prepare DIO to read data
321  TM1637_DIO_HIGH();
322  TM1637_DIO_INPUT();
323  TM1637_PIN_DELAY();
324 
325  // Data is shifted out by the TM1637 on the CLK falling edge
326  for (uint8_t bit = 0; bit < 8; bit++) {
327  TM1637_CLK_HIGH();
328  TM1637_PIN_DELAY();
329 
330  // Read next bit
331  retval <<= 1;
332  if (TM1637_DIO_READ()) {
333  retval |= 0x01;
334  }
335 
336  TM1637_CLK_LOW();
337  TM1637_PIN_DELAY();
338  }
339 
340  // Return DIO to output mode
341  TM1637_DIO_OUTPUT();
342  TM1637_PIN_DELAY();
343 
344  // Dummy ACK
345  TM1637_DIO_LOW();
346  TM1637_PIN_DELAY();
347 
348  TM1637_CLK_HIGH();
349  TM1637_PIN_DELAY();
350  TM1637_CLK_LOW();
351  TM1637_PIN_DELAY();
352 
353  TM1637_DIO_HIGH();
354  TM1637_PIN_DELAY();
355 
356  return retval;
357 }
